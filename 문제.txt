var results = document.getElementById('results');
var r = new XMLHttpRequest();
r.open("GET", "http://www.filltext.com?rows=3&f={firstName}", true);
r.onreadystatechange = function () {
  if (r.readyState != 4 || r.status != 200) return;
  var data = JSON.parse(r.responseText);
  for (i=0;i<data.length;i++){
        results.innerHTML += '<li>'+data[i].f+'</li>'
  }
};
r.send();
_____________________________________________________________________
. 위 코드는 filltext.com에서 json 데이터 3개를 배열로 가져오고 results id를 가진 list에 append하는 코드입니다.
버튼을 누를 때마다 실행되도록 하고 결과가 문서에 출력 되도록 하세요.

function appendDataToTable() {
    var url = "http://www.filltext.com/?callback=?";
  $.getJSON( url, {
    'rows': 4,
    'fname': '{firstName}',
    'lname': '{lastName}',
    'tel': '{phone|format}',
  })
  .done(function( data ) {
    $.each( data, function( i, item ) {
        var html = 
            "<td>" + item.fname + "</td>" +
            "<td>" + item.lname + "</td>" +
            "<td>" + item.tel + "</td>";
          $("<tr/>").html(html).appendTo("#records");
    });
  });
}
_____________________________________________________________________
